#include <stdio.h>
intmain()
{
    intx, y, sum; // Declare variables for two integers and their sum
    
    // Prompt user for input and store in 'x'
    printf("\nInput the first integer: ");
    scanf("%d", &x);
    
    // Prompt user for input and store in 'y'
    printf("\nInput the second integer: ");
    scanf("%d", &y);
    
    sum = x + y; // Calculate the sum of 'x' and 'y'
    
    // Print the sum
    printf("\nSum of the above two integers = %d\n", sum);
    
    return0; // Indicate successful execution
}

  • int x, y, sum;: 宣告三個整數變數，用來存儲兩個整數和它們的總和。

  • printf("\n輸入第一個整數：");: 使用 printf 函式輸出提示信息，要求用戶輸入第一個整數。

  • scanf("%d", &x);: 使用 scanf 函式從用戶輸入中讀取一個整數，並將其存儲在變數 x 中。

  • printf("\n輸入第二個整數：");: 使用 printf 函式輸出提示信息，要求用戶輸入第二個整數。

  • scanf("%d", &y);: 使用 scanf 函式從用戶輸入中讀取另一個整數，並將其存儲在變數 y 中。

  • sum = x + y;: 計算變數 x 和 y 的總和，並將結果存儲在變數 sum 中。

  • printf("\n以上兩個整數的總和 = %d\n", sum);: 使用 printf 函式輸出計算得到的總和。

  • return 0;: 主函式正常結束，返回 0。
